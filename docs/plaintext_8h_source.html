475 SEAL_NODISCARD inline bool operator==(const Plaintext &compare) const
476 {
477 std::size_t sig_coeff_count = significant_coeff_count();
478 std::size_t sig_coeff_count_compare = compare.significant_coeff_count();
479 bool parms_id_compare = (is_ntt_form() && compare.is_ntt_form() && (parms_id_ == compare.parms_id_)) ||
480 (!is_ntt_form() && !compare.is_ntt_form());
481 return parms_id_compare && (sig_coeff_count == sig_coeff_count_compare) &&
482 std::equal(
483 data_.cbegin(), data_.cbegin() + sig_coeff_count, compare.data_.cbegin(),
484 compare.data_.cbegin() + sig_coeff_count) &&
485 std::all_of(data_.cbegin() + sig_coeff_count, data_.cend(), util::is_zero<pt_coeff_type>) &&
486 std::all_of(
487 compare.data_.cbegin() + sig_coeff_count, compare.data_.cend(), util::is_zero<pt_coeff_type>) &&
488 util::are_close(scale_, compare.scale_);
489 }
