64#define SEAL_DIVIDE_UINT128_UINT64(numerator, denominator, result) \
65 do \
66 { \
67 uint128_t n, q; \
68 n = (static_cast<uint128_t>(numerator[1]) << 64) | (static_cast<uint128_t>(numerator[0])); \
69 q = n / denominator; \
70 n -= q * denominator; \
71 numerator[0] = static_cast<std::uint64_t>(n); \
72 numerator[1] = 0; \
73 result[0] = static_cast<std::uint64_t>(q); \
74 result[1] = static_cast<std::uint64_t>(q >> 64); \
75 } while (false)
