444 template <
445 typename T, typename = std::enable_if_t<
446 std::is_same<std::remove_cv_t<T>, double>::value ||
447 std::is_same<std::remove_cv_t<T>, std::complex<double>>::value>>
448 void encode_internal(
449 const T *values, std::size_t values_size, parms_id_type parms_id, double scale, Plaintext &destination,
450 MemoryPoolHandle pool) const
451 {
452 // Verify parameters.
453 auto context_data_ptr = context_.get_context_data(parms_id);
454 if (!context_data_ptr)
455 {
456 throw std::invalid_argument("parms_id is not valid for encryption parameters");
457 }
458 if (!values && values_size > 0)
459 {
460 throw std::invalid_argument("values cannot be null");
461 }
462 if (values_size > slots_)
463 {
464 throw std::invalid_argument("values_size is too large");
465 }
466 if (!pool)
467 {
468 throw std::invalid_argument("pool is uninitialized");
469 }
470
471 auto &context_data = *context_data_ptr;
472 auto &parms = context_data.parms();
473 auto &coeff_modulus = parms.coeff_modulus();
474 std::size_t coeff_modulus_size = coeff_modulus.size();
475 std::size_t coeff_count = parms.poly_modulus_degree();
476
477 // Quick sanity check
478 if (!util::product_fits_in(coeff_modulus_size, coeff_count))
479 {
480 throw std::logic_error("invalid parameters");
481 }
482
483 // Check that scale is positive and not too large
484 if (scale <= 0 || (static_cast<int>(log2(scale)) + 1 >= context_data.total_coeff_modulus_bit_count()))
485 {
486 throw std::invalid_argument("scale out of bounds");
487 }
488
489 auto ntt_tables = context_data.small_ntt_tables();
490
491 // values_size is guaranteed to be no bigger than slots_
492 std::size_t n = util::mul_safe(slots_, std::size_t(2));
493
494 auto conj_values = util::allocate<std::complex<double>>(n, pool, 0);
495 for (std::size_t i = 0; i < values_size; i++)
496 {
497 conj_values[matrix_reps_index_map_[i]] = values[i];
498 // TODO: if values are real, the following values should be set to zero, and multiply results by 2.
499 conj_values[matrix_reps_index_map_[i + slots_]] = std::conj(values[i]);
500 }
501 double fix = scale / static_cast<double>(n);
502 fft_handler_.transform_from_rev(conj_values.get(), util::get_power_of_two(n), inv_root_powers_.get(), &fix);
503
504 double max_coeff = 0;
505 for (std::size_t i = 0; i < n; i++)
506 {
507 max_coeff = std::max<>(max_coeff, std::fabs(conj_values[i].real()));
508 }
509 // Verify that the values are not too large to fit in coeff_modulus
510 // Note that we have an extra + 1 for the sign bit
511 // Don't compute logarithmis of numbers less than 1
512 int max_coeff_bit_count = static_cast<int>(std::ceil(std::log2(std::max<>(max_coeff, 1.0)))) + 1;
513 if (max_coeff_bit_count >= context_data.total_coeff_modulus_bit_count())
514 {
515 throw std::invalid_argument("encoded values are too large");
516 }
517
518 double two_pow_64 = std::pow(2.0, 64);
519
520 // Resize destination to appropriate size
521 // Need to first set parms_id to zero, otherwise resize
522 // will throw an exception.
523 destination.parms_id() = parms_id_zero;
524 destination.resize(util::mul_safe(coeff_count, coeff_modulus_size));
525
526 // Use faster decomposition methods when possible
527 if (max_coeff_bit_count <= 64)
528 {
529 for (std::size_t i = 0; i < n; i++)
530 {
531 double coeffd = std::round(conj_values[i].real());
532 bool is_negative = std::signbit(coeffd);
533
534 std::uint64_t coeffu = static_cast<std::uint64_t>(std::fabs(coeffd));
535
536 if (is_negative)
537 {
538 for (std::size_t j = 0; j < coeff_modulus_size; j++)
539 {
540 destination[i + (j * coeff_count)] = util::negate_uint_mod(
541 util::barrett_reduce_64(coeffu, coeff_modulus[j]), coeff_modulus[j]);
542 }
543 }
544 else
545 {
546 for (std::size_t j = 0; j < coeff_modulus_size; j++)
547 {
548 destination[i + (j * coeff_count)] = util::barrett_reduce_64(coeffu, coeff_modulus[j]);
549 }
550 }
551 }
552 }
553 else if (max_coeff_bit_count <= 128)
554 {
555 for (std::size_t i = 0; i < n; i++)
556 {
557 double coeffd = std::round(conj_values[i].real());
558 bool is_negative = std::signbit(coeffd);
559 coeffd = std::fabs(coeffd);
560
561 std::uint64_t coeffu[2]{ static_cast<std::uint64_t>(std::fmod(coeffd, two_pow_64)),
562 static_cast<std::uint64_t>(coeffd / two_pow_64) };
563
564 if (is_negative)
565 {
566 for (std::size_t j = 0; j < coeff_modulus_size; j++)
567 {
568 destination[i + (j * coeff_count)] = util::negate_uint_mod(
569 util::barrett_reduce_128(coeffu, coeff_modulus[j]), coeff_modulus[j]);
570 }
571 }
572 else
573 {
574 for (std::size_t j = 0; j < coeff_modulus_size; j++)
575 {
576 destination[i + (j * coeff_count)] = util::barrett_reduce_128(coeffu, coeff_modulus[j]);
577 }
578 }
579 }
580 }
581 else
582 {
583 // Slow case
584 auto coeffu(util::allocate_uint(coeff_modulus_size, pool));
585 for (std::size_t i = 0; i < n; i++)
586 {
587 double coeffd = std::round(conj_values[i].real());
588 bool is_negative = std::signbit(coeffd);
589 coeffd = std::fabs(coeffd);
590
591 // We are at this point guaranteed to fit in the allocated space
592 util::set_zero_uint(coeff_modulus_size, coeffu.get());
593 auto coeffu_ptr = coeffu.get();
594 while (coeffd >= 1)
595 {
596 *coeffu_ptr++ = static_cast<std::uint64_t>(std::fmod(coeffd, two_pow_64));
597 coeffd /= two_pow_64;
598 }
599
600 // Next decompose this coefficient
601 context_data.rns_tool()->base_q()->decompose(coeffu.get(), pool);
602
603 // Finally replace the sign if necessary
604 if (is_negative)
605 {
606 for (std::size_t j = 0; j < coeff_modulus_size; j++)
607 {
608 destination[i + (j * coeff_count)] = util::negate_uint_mod(coeffu[j], coeff_modulus[j]);
609 }
610 }
611 else
612 {
613 for (std::size_t j = 0; j < coeff_modulus_size; j++)
614 {
615 destination[i + (j * coeff_count)] = coeffu[j];
616 }
617 }
618 }
619 }
620
621 // Transform to NTT domain
622 for (std::size_t i = 0; i < coeff_modulus_size; i++)
623 {
624 util::ntt_negacyclic_harvey(destination.data(i * coeff_count), ntt_tables[i]);
625 }
626
627 destination.parms_id() = parms_id;
628 destination.scale() = scale;
629 }

631 template <
632 typename T, typename = std::enable_if_t<
633 std::is_same<std::remove_cv_t<T>, double>::value ||
634 std::is_same<std::remove_cv_t<T>, std::complex<double>>::value>>
635 void decode_internal(const Plaintext &plain, T *destination, MemoryPoolHandle pool) const
636 {
637 // Verify parameters.
638 if (!is_valid_for(plain, context_))
639 {
640 throw std::invalid_argument("plain is not valid for encryption parameters");
641 }
642 if (!plain.is_ntt_form())
643 {
644 throw std::invalid_argument("plain is not in NTT form");
645 }
646 if (!destination)
647 {
648 throw std::invalid_argument("destination cannot be null");
649 }
650 if (!pool)
651 {
652 throw std::invalid_argument("pool is uninitialized");
653 }
654
655 auto &context_data = *context_.get_context_data(plain.parms_id());
656 auto &parms = context_data.parms();
657 std::size_t coeff_modulus_size = parms.coeff_modulus().size();
658 std::size_t coeff_count = parms.poly_modulus_degree();
659 std::size_t rns_poly_uint64_count = util::mul_safe(coeff_count, coeff_modulus_size);
660
661 auto ntt_tables = context_data.small_ntt_tables();
662
663 // Check that scale is positive and not too large
664 if (plain.scale() <= 0 ||
665 (static_cast<int>(log2(plain.scale())) >= context_data.total_coeff_modulus_bit_count()))
666 {
667 throw std::invalid_argument("scale out of bounds");
668 }
669
670 auto decryption_modulus = context_data.total_coeff_modulus();
671 auto upper_half_threshold = context_data.upper_half_threshold();
672 int logn = util::get_power_of_two(coeff_count);
673
674 // Quick sanity check
675 if ((logn < 0) || (coeff_count < SEAL_POLY_MOD_DEGREE_MIN) || (coeff_count > SEAL_POLY_MOD_DEGREE_MAX))
676 {
677 throw std::logic_error("invalid parameters");
678 }
679
680 double inv_scale = double(1.0) / plain.scale();
681
682 // Create mutable copy of input
683 auto plain_copy(util::allocate_uint(rns_poly_uint64_count, pool));
684 util::set_uint(plain.data(), rns_poly_uint64_count, plain_copy.get());
685
686 // Transform each polynomial from NTT domain
687 for (std::size_t i = 0; i < coeff_modulus_size; i++)
688 {
689 util::inverse_ntt_negacyclic_harvey(plain_copy.get() + (i * coeff_count), ntt_tables[i]);
690 }
691
692 // CRT-compose the polynomial
693 context_data.rns_tool()->base_q()->compose_array(plain_copy.get(), coeff_count, pool);
694
695 // Create floating-point representations of the multi-precision integer coefficients
696 double two_pow_64 = std::pow(2.0, 64);
697 auto res(util::allocate<std::complex<double>>(coeff_count, pool));
698 for (std::size_t i = 0; i < coeff_count; i++)
699 {
700 res[i] = 0.0;
701 if (util::is_greater_than_or_equal_uint(
702 plain_copy.get() + (i * coeff_modulus_size), upper_half_threshold, coeff_modulus_size))
703 {
704 double scaled_two_pow_64 = inv_scale;
705 for (std::size_t j = 0; j < coeff_modulus_size; j++, scaled_two_pow_64 *= two_pow_64)
706 {
707 if (plain_copy[i * coeff_modulus_size + j] > decryption_modulus[j])
708 {
709 auto diff = plain_copy[i * coeff_modulus_size + j] - decryption_modulus[j];
710 res[i] += diff ? static_cast<double>(diff) * scaled_two_pow_64 : 0.0;
711 }
712 else
713 {
714 auto diff = decryption_modulus[j] - plain_copy[i * coeff_modulus_size + j];
715 res[i] -= diff ? static_cast<double>(diff) * scaled_two_pow_64 : 0.0;
716 }
717 }
718 }
719 else
720 {
721 double scaled_two_pow_64 = inv_scale;
722 for (std::size_t j = 0; j < coeff_modulus_size; j++, scaled_two_pow_64 *= two_pow_64)
723 {
724 auto curr_coeff = plain_copy[i * coeff_modulus_size + j];
725 res[i] += curr_coeff ? static_cast<double>(curr_coeff) * scaled_two_pow_64 : 0.0;
726 }
727 }
728
729 // Scaling instead incorporated above; this can help in cases
730 // where otherwise pow(two_pow_64, j) would overflow due to very
731 // large coeff_modulus_size and very large scale
732 // res[i] = res_accum * inv_scale;
733 }
734
735 fft_handler_.transform_to_rev(res.get(), logn, root_powers_.get());
736
737 for (std::size_t i = 0; i < slots_; i++)
738 {
739 destination[i] = from_complex<T>(res[static_cast<std::size_t>(matrix_reps_index_map_[i])]);
740 }
741 }
