351 SEAL_NODISCARD inline std::streamoff save_size(
352 compr_mode_type compr_mode = Serialization::compr_mode_default) const
353 {
354 std::size_t coeff_modulus_total_size =
355 coeff_modulus_.empty()
356 ? std::size_t(0)
357 : util::safe_cast<std::size_t>(coeff_modulus_[0].save_size(compr_mode_type::none));
358 coeff_modulus_total_size = util::mul_safe(coeff_modulus_total_size, coeff_modulus_.size());
359
360 std::size_t members_size = Serialization::ComprSizeEstimate(
361 util::add_safe(
362 sizeof(scheme_),
363 sizeof(std::uint64_t), // poly_modulus_degree_
364 sizeof(std::uint64_t), // coeff_modulus_size
365 coeff_modulus_total_size,
366 util::safe_cast<std::size_t>(plain_modulus_.save_size(compr_mode_type::none))),
367 compr_mode);
368
369 return util::safe_cast<std::streamoff>(util::add_safe(sizeof(Serialization::SEALHeader), members_size));
370 }
