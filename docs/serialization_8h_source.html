144 SEAL_NODISCARD static bool IsCompatibleVersion(const SEALHeader &header) noexcept
145 {
146 // Exact same version
147 if (header.version_major == SEAL_VERSION_MAJOR && header.version_minor == SEAL_VERSION_MINOR)
148 {
149 return true;
150 }
151
152 // Different major versions not supported
153 if (header.version_major != SEAL_VERSION_MAJOR && header.version_major != 3)
154 {
155 return false;
156 }
157
158 // Support Microsoft SEAL 3.4 and above
159 if (header.version_major == 3 && header.version_minor >= 4)
160 {
161 return true;
162 }
163
164 return false;
165 }

172 SEAL_NODISCARD static bool IsValidHeader(const SEALHeader &header) noexcept
173 {
174 if (header.magic != seal_magic)
175 {
176 return false;
177 }
178 if (header.header_size != seal_header_size)
179 {
180 return false;
181 }
182 if (!IsCompatibleVersion(header))
183 {
184 return false;
185 }
186 if (!IsSupportedComprMode(static_cast<uint8_t>(header.compr_mode)))
187 {
188 return false;
189 }
190 return true;
191 }
