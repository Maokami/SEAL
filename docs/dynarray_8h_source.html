425 inline void reserve(std::size_t capacity)
426 {
427 std::size_t copy_size = std::min<>(capacity, size_);
428
429 // Create new allocation and copy over value
430 auto new_data(util::allocate<T>(capacity, pool_));
431 std::copy_n(cbegin(), copy_size, new_data.get());
432 std::swap(data_, new_data);
433
434 // Set the coeff_count and capacity
435 capacity_ = capacity;
436 size_ = copy_size;

457 inline void resize(std::size_t size, bool fill_zero = true)
458 {
459 if (size <= capacity_)
460 {
461 // Are we changing size to bigger within current capacity?
462 // If so, need to set top terms to zero
463 if (size > size_ && fill_zero)
464 {
465 std::fill(end(), begin() + size, T(0));
466 }
467
468 // Set the size
469 size_ = size;
470
471 return;
472 }
473
474 // At this point we know for sure that size_ <= capacity_ < size so need
475 // to reallocate to bigger
476 auto new_data(util::allocate<T>(size, pool_));
477 std::copy(cbegin(), cend(), new_data.get());
478 if (fill_zero)
479 {
480 std::fill(new_data.get() + size_, new_data.get() + size, T(0));
481 }
482 std::swap(data_, new_data);
483
484 // Set the coeff_count and capacity
485 capacity_ = size;
486 size_ = size;
487 }

652 void save_members(std::ostream &stream) const
653 {
654 auto old_except_mask = stream.exceptions();
655 try
656 {
657 // Throw exceptions on std::ios_base::badbit and std::ios_base::failbit
658 stream.exceptions(std::ios_base::badbit | std::ios_base::failbit);
659
660 std::uint64_t size64 = size_;
661 stream.write(reinterpret_cast<const char *>(&size64), sizeof(std::uint64_t));
662 if (size_)
663 {
664 stream.write(
665 reinterpret_cast<const char *>(cbegin()),
666 util::safe_cast<std::streamsize>(util::mul_safe(size_, sizeof(T))));
667 }
668 }
669 catch (const std::ios_base::failure &)
670 {
671 stream.exceptions(old_except_mask);
672 throw std::runtime_error("I/O error");
673 }
674 catch (...)
675 {
676 stream.exceptions(old_except_mask);
677 throw;
678 }
679 stream.exceptions(old_except_mask);
680 }

682 void load_members(std::istream &stream, SEAL_MAYBE_UNUSED SEALVersion version, std::size_t in_size_bound)
683 {
684 auto old_except_mask = stream.exceptions();
685 try
686 {
687 // Throw exceptions on std::ios_base::badbit and std::ios_base::failbit
688 stream.exceptions(std::ios_base::badbit | std::ios_base::failbit);
689
690 std::uint64_t size64 = 0;
691 stream.read(reinterpret_cast<char *>(&size64), sizeof(std::uint64_t));
692
693 // Check (optionally) that the size in the metadata does not exceed
694 // in_size_bound
695 if (in_size_bound && util::unsigned_gt(size64, in_size_bound))
696 {
697 throw std::logic_error("unexpected size");
698 }
699
700 // Set new size; this is potentially unsafe if size64 was not checked
701 // against expected_size
702 resize(util::safe_cast<std::size_t>(size64));
703
704 // Read data
705 if (size_)
706 {
707 stream.read(
708 reinterpret_cast<char *>(begin()),
709 util::safe_cast<std::streamsize>(util::mul_safe(size_, sizeof(T))));
710 }
711 }
712 catch (const std::ios_base::failure &)
713 {
714 stream.exceptions(old_except_mask);
715 throw std::runtime_error("I/O error");
716 }
717 catch (...)
718 {
719 stream.exceptions(old_except_mask);
720 throw;
721 }
722 stream.exceptions(old_except_mask);
723 }
