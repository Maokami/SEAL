337 SEAL_NODISCARD inline ct_coeff_type *data(std::size_t poly_index)
338 {
339 auto poly_uint64_count = util::mul_safe(poly_modulus_degree_, coeff_modulus_size_);
340 if (poly_uint64_count == 0)
341 {
342 return nullptr;
343 }
344 if (poly_index >= size_)
345 {
346 throw std::out_of_range("poly_index must be within [0, size)");
347 }
348 return data_.begin() + util::safe_cast<std::size_t>(util::mul_safe(poly_index, poly_uint64_count));
349 }
