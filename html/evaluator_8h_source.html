1335 inline void conjugate_internal(
1336 Ciphertext &encrypted, const GaloisKeys &galois_keys, MemoryPoolHandle pool) const
1337 {
1338 // Verify parameters.
1339 auto context_data_ptr = context_.get_context_data(encrypted.parms_id());
1340 if (!context_data_ptr)
1341 {
1342 throw std::invalid_argument("encrypted is not valid for encryption parameters");
1343 }
1344
1345 // Extract encryption parameters.
1346 auto &context_data = *context_data_ptr;
1347 if (!context_data.qualifiers().using_batching)
1348 {
1349 throw std::logic_error("encryption parameters do not support batching");
1350 }
1351
1352 auto galois_tool = context_data.galois_tool();
1353
1354 // Perform rotation and key switching
1355 apply_galois_inplace(encrypted, galois_tool->get_elt_from_step(0), galois_keys, std::move(pool));
1356 }
